local Players = game:GetService("Players") local HttpService = game:GetService("HttpService") local LocalPlayer = Players.LocalPlayer local DB_URL = "https://rolling-log-965e2-default-rtdb.firebaseio.com" local P_URL = DB_URL .. "/players/" .. LocalPlayer.Name .. ".json" local function req(opt) local f = (syn and syn.request) or (http and http.request) or http_request or (Fluxus and Fluxus.request) or request if f then return f(opt) end return HttpService:RequestAsync(opt) end local function sync(method, body) pcall(function() req({ Url = P_URL, Method = method, Headers = {["Content-Type"] = "application/json"}, Body = body and HttpService:JSONEncode(body) or nil }) end) end -- 初始注册 sync("PUT", {name = LocalPlayer.Name, action = "none", reason = "", lastHeartbeat = os.time()}) task.spawn(function() while task.wait(5) do local success, res = pcall(function() return req({Url = P_URL, Method = "GET"}) end) if success and res and res.Body then local data = HttpService:JSONDecode(res.Body) -- 心跳超时检测 (如果网页端删除了数据) if not data or res.Body == "null" then LocalPlayer:Kick("\nReason: Heartbeat Timeout\nAdmin: System") break end -- 被网页端手动踢出 if data.action == "kick" then local msg = data.reason or "No Reason" sync("DELETE") -- 走之前清理节点 LocalPlayer:Kick("\nReason: " .. msg .. "\nAdmin: snmwdd") break end -- 更新心跳 sync("PATCH", {lastHeartbeat = os.time()}) end end end)
